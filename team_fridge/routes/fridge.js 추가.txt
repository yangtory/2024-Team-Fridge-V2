router.post("/shopmemo", (req, res) => {
  const todo1 = req.body.todo1;
  const todo2 = req.body.todo2;

  const params = [todo1, todo2];
  const sql = "INSERT INTO tbl_shopping (s_name, s_quan) VALUES (?, ?)";

  connection.query(sql, params, (err, result) => {
    if (err) {
      return res.json(err);
    }
  });
});
